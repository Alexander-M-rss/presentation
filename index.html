Ten Essential Linux Commands

find 
 $ find . -type f -name 'nameofthefile.ext'
 $ find . -type f -perm 0777 -print
 $ find /tmp -type d -empty
 $ find . -type f -name 'nameofthefile.ext' -ls
 $ man find
grep 
 $ grep "literal_string" filename
 $ grep "this" demo_*.*
 $ grep "sample.*empty" demo_file
sed 
 $ sed -i '' s/test/beta/g /home/username/sample.txt
 $ sed -i '1d' /home/username/sample.txt
 $ sed -i '4,10d' /home/username/sample.txt
 $ sed '/^$/d' /home/username/sample.txt
head 
 $ head -n 6 sample.txt
 $ head -c 13 sample.txt
 $ head -n -5 sample.txt
 $ head -c -30 sample.txt
 $ head -n 4 -q *.*
locate 
 $ locate *apache*.txt
 $ locate -c *apache*.txt
 $ locate -e *apache*.txt
 $ locate --regex -i "(\.mp4|\.avi)"
alias 
 $ alias myrs='sudo service mysql restart'
 $ unalias myrs='sudo service mysql restart'
 $ alias alias-list='cat ~/.bash_aliases'
 $ alias alias-edit='nano ~/.bash_aliases'
 $ alias ara='cd ~/repo/async-race-api'
tar 
 $ tar -cvf sample-22-08-15.tar /home/sampleuser/
 $ tar -cvfz photos-22-08-15.tar.gz /home/user/photos
 $ tar -xvf photos-22-08-15.tar.gz
 $ tar -xvf photos-22-08-15.tar.gz /home/user/photos
scp 
 $ scp mytb.tar.gz user@server.com:/home/user/folder
 $ scp user@server.com:/home/user/dir ./mytb.tar.gz
 $ scp -i key.pem mytb.tar.gz user@server.com:/folder
df 
 $ df -T -h
 $ df -h -t ext4 --output=source,size,pcent  
 $ df -x ext4 -h
lsof 
 $ lsof
 $ sudo lsof -u username
 $ sudo lsof -i -P -n | grep LISTEN
